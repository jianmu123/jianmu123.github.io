数组中常用的方法

  • 方法的作用和含义
  • 方法的实参（类型和含义）
  • 方法的返回值
  • 原来的数组是否会发生改变

1：实现数组增删改的方法

这一部分方法都会改变原数组

push

push:向数组末尾增加内容

参数： 多个任意类型

返回值：返回的是新增的数组长度，原来数组会改变。

let ary=[10,20];
let res=ary.push(30,'aa',{
    name:"珠峰"
});
console.log(res,ary);
//5（新数组长度）;
0: 10
1: 20
2: 30
3: "aa"
4: {name: "珠峰"}
length: 5
也可以使用原生js增加（但是一次只能加1个），如：
ary[ary.length]=40;//在数组末尾添加
console.log(ary);

unshift

unshift：向数组开始位置增加内容

参数：多个任意类型

返回值： 新增数组长度，会改变原数组

let ary=[10,20];
let res=ary.unshift(30,'aa');
console.log(res,ary);
//4;
0: 30
1: "aa"
2: 10
3: 20
length: 4

shift

shift:删除数组第一项

参数：没有

返回值：被删除的那 一项，会改变原数组

let ary=[10,20];
let res=ary.shift();
console.log(res,ary);
//10;
0: 20
length: 1

pop

pop:删除数组最后一项

参数：无

返回值：被删除的那 一项，会改变原数组

let ary=[10,20];
let res=ary.pop();
console.log(res,ary);
//20;
0: 10
length: 1

splice实现数组的增加、删除、修改

splice:实现删除

参数：n,m 都是数字，从索引n开始删除m个元素（如果不写m，则删除到末尾）

返回值： 把删除的部分用新数组存储起来返回

let ary[10,20,30,40,50,60,70,80,90];
let res=ary.splice(2,4);
console.log(res,ary);// [30，40，50，60]  [10，20，70，80，90]

splice:实现增加，修改

参数：n,m,x ：从索引为n开始删除m个元素，然后用x占用那个位置

​ n,0,x: 从索引n开始，，一个都不删，把x放到索引n前面

返回值：把删除的部分用新数组村粗起来返回

splice:实现向数组末尾增加： splice(ary.length,0,'AA);

splice:实现向数组开头追加： splice(0,0,'BB);

2:数组的查询和拼接

此组的方法：原来数组不会改变

slice

slice 实现数组查询

参数： n,m 从索引n开始，找到索引为m的地方（不含m这一项）,m不写则找到末尾

返回值： 找到的内容以一个新数组的形式返回

let ary =[10,20,30,40,50];
let res=ary.slice(1,3);
console.log(res);// [20,30]
//数组的克隆
let res=ary.slice(0);
console.log(res);//[10,20,30,40,50]

concat

concat：实现数组拼接

参数：任意类型值

返回值： 拼接后的新数组（原数组不变）

let ary =[10,20,30,40,50];
let res=ary.concat('珠峰');
console.log(res);// [10,20,30,40,50，'珠峰']

3：把数组转换为字符串

原数组不变

toString

toString：把数组转换为字符串

参数：无

返回值： 转换后的新字符串，每一项用逗号分隔（原数组不变）

let ary =[10,20,30];
let res=ary.toString();
console.log(res);//"10,20,30"

join

join：把数组转换为字符串

参数：指定的分隔符（不指定则是逗号，和tostring一样）

返回值： 转换后的新字符串，每一项用指定分隔符分隔（原数组不变）

let ary =[10,20,30];
let res=ary.join();
console.log(res);//"10,20,30"

let res=ary.join("");
console.log(res);//"102030"

let res=ary.join("+");
console.log(res);//"10+20+30"
console.log(eval(res));// =>60 因为eval可以把字符串转为js表达式执行

4:检测数组是否含有某一项

indexOf/lastIndexOf(IE6~8不兼容)

indexOf/lastIndexOf: 检测当前项在数组中第一次或最后一次出现的位置的索引值。

参数： 要检测的内容

返回值： 这一项出现的位置索引值（数字），如果数组没有这一项返回-1

let ary=[10,20,30,10,20,30];
console.log(ary.indexOf(20));//1
console.log(ary.lastIndexOf(20));//4
可以使用该办法验证是否包含一个东西
if(ary.indexOf('珠峰')===-1){
console.log("不包含")
}else{
console.log("包含")
}

或则使用es6的inludes方法(ie6~8不兼容)
if(ary.includes("珠峰")){
    //如果包含返回true
}

5：数组的排序或则排列

reverse

reverse: 把数组倒过来排列

参数： 无

返回值：排列后的新数组，原数组改变

let ary=[12,15,9,28,10,22];
ary.reverse;
console.log(ary);// [22,10,28,9,15,12]

sort

sort: 实现数组的排序

参数：可以没有也可以是函数，但是如果不传递参数，是无法处理10以上的数字

返回值：排列后的新数组，原数组改变

let ary=[7,8,5,1,6,4];
ary.sort();
console.log(ary);//[1, 4, 5, 6, 7, 8]

注意：sort如果不传递参数，是无法处理10以上的数字
let ary=[7,18,15,21,26,9];
ary.sort();
console.log(ary);//[15, 18, 21, 26, 7, 9] 它是按照第一个开头数字排序
修改：
let ary=[7,18,15,21,26,9];
ary.sort((a,b)=>{
    //a,b是数组相邻两项
    //return a-b:从小到大排
    //return b-a:从大到小排
    return a-b;
});
console.log(ary);//[7, 9, 15, 18, 21, 26]

6：遍历数组的每一项的方法

foreach map filter find reduce some every 、、、

foreach: 遍历数组中的每一项内容

参数：函数

返回值： 没有

原数组不变

基于原生js的循环可以实现

如：

let ary=[10,20,30,40];
for(var i=0;i<ary.length;i++){
    console.log('索引:'+i+'内容:'+ary[i]);
}

下面使用foreach方法

let ary=[10,20,30,40];
ary.forEach(function(item,index){
    //数组有多少项,函数就会被默认执行多少次
    //每一次执行函数，item就是当前数组要操作的这一项（相当于上面原生js方法的ary[i]），index是当前项的索引（相当于上面原生js方法的i）
     console.log('索引:'+index+'内容:'+item);
});

7案列：数组去重

 <script>
      /*
      注意数组塌陷问题
       */
      /* 重建数组去重函数 */
      function unique(ary) {
        let obj = {}; //创建一个对象来存取item：item
        for (let i = 0; i < ary.length; i++) {
          let item = ary[i];
          if (obj[item] !== undefined) {
            //代表这个数是重复的
            ary[i] = ary[ary.length - 1]; //让这个重复的数被最后一个数替换
            ary.length--;
            i--; //替换后继续检测这个位置的数，这里是数组塌陷问题
            continue;
          }
          obj[item] = item;
        }
        return ary;
      }
      let aa = [1, 2, 3, 1, 2, 3];
      aa = unique(aa);
      console.log(aa);
